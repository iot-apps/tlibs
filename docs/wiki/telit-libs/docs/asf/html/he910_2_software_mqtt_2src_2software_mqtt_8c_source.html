107 /******************************************************************************
108  * APLICAÇÃO
109  ******************************************************************************/
110 static void softwareMqttTask(void *params)
111 {
112  LEDS_Init();
113  SERIAL_Init();
114 
115 #ifdef VERBOSE
116  SERIAL_Print("\r\n");
117  SERIAL_Print("******************************\r\n");
118  SERIAL_Print("* APLICAÇÃO DE EXEMPLO TELIT *\r\n");
119  SERIAL_Print("* SOFTWARE MQTT *\r\n");
120  SERIAL_Print("******************************\r\n");
121  SERIAL_Print("\r\n");
122 #endif
123 
124  /**********************************************
125  * Aguarda o Wifi ser iniciado com sucesso.
126  **********************************************/
127  SERIAL_Print("Iniciando...");
128  if ( GSM_Init(DEFAULT_GSM_APN, DEFAULT_GSM_PROTOCOL) )
129  {
130 #ifdef VERBOSE
131  SERIAL_Println("OK");
132 #endif
133  }
134  else
135  {
136 #ifdef VERBOSE
137  SERIAL_Println("ER");
138 #endif
139  }
140 
141  /**********************************************
142  * O loop da aplicação principal é responsável
143  * pelo gerenciamento das conexões WIFI e TCP.
144  **********************************************/
145  for(;;)
146  {
147 #ifdef SHOW_HEAP_FREE
148  {
149  char* toPrint = (char*)pvPortMalloc(32);
150 
151  sprintf(toPrint, "Heap Free: %d", xPortGetFreeHeapSize());
152  SERIAL_Println(toPrint);
153 
154  vPortFree(toPrint);
155  }
156 #endif
157 
158  if ( !GSM_IsConnected() )
159  {
160 #ifdef VERBOSE
161  SERIAL_Print("Conectando...");
162 #endif
163 
164  if ( GSM_Connect() )
165  {
166  #ifdef VERBOSE
167  SERIAL_Println("OK");
168  #endif
169  }
170  else
171  {
172  #ifdef VERBOSE
173  SERIAL_Println("ER");
174  #endif
175  }
176  }
177  else
178  {
179  if (! MQTT_IsConnected())
180  {
181  gu8_cid = MQTT_Connect(MQTT_DEFAULT_ADDRESS, MQTT_DEFAULT_PORT, MQTT_DEFAULT_CLIENT_ID, MQTT_DEFAULT_USR, MQTT_DEFAULT_PWD);
182  }
183  else if ( gu8_cid != 0xFF )
184  {
185 #ifdef VERBOSE
186  SERIAL_Println("MQTT Connect OK");
187 
188 #endif
189  sendMessages();
190  }
191  else
192  {
193 #ifdef VERBOSE
194  SERIAL_Println("MQTT Connect ER");
195 #endif
196  }
197  }
198 
199 
200  vTaskDelay(SOFTWAREMQTT_TASK_DELAY);
201  }
202 
203  MQTT_Deinit();
204  vTaskDelete(NULL);
205 }
