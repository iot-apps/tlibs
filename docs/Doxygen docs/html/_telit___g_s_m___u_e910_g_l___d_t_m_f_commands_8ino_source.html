16  * INSTRUÇÕES DE UTILIZAÇÃO DO EXEMPLO
17  *
18  * Esse exemplo foi criado para utilização do kit EVK2 da Telit integrado com o arduino M0 PRO.
19  * Para correto funcionamento a conexão das placas devem ser de acordo com a seguinte pinangem:
20  * Pino Arduino M0 Pro Pino EVK2

22  * 10 C104/RXD
23  * 11 C103/TXD
24  * 09 RESET
25  *
26  *
27  * INSTRUÇÕES PARA UTILIZAÇÃO DE ARQUIVOS DE AUDIO
28  *
29  * Para o uso de audio pre-gravados pelo modem da telit para respostas, é necessario que os arquivos de audio
30  * esteja armazenado na memoria do modulo no formato wav ou pcm.
31  * O carregamento do arquivo é realizado através do aplicativo para windows TELIT AT CONTROLLER, que pode ser encontrado
32  * junto aos drivers do kit EVK2 em: https://www.telit.com/evkevb-drivers/
33  *
34  * Visite site https://www.????????????/ para mais informações sobre como utilizar adequadamente e o app Telit AT Controller
35  * e sobre duvidas da conexão entre o computador e o EVK2
36  *
37  * CARREGAMENTO DE ARQUIVOS DE AUDIO TELIT AT CONTROLLER
38  *
39  * Através do terminal do Telit AT Controller, siga os seguintes passos para o carregamento do arquivo.
40  *
41  * Insira o comando abaixo.
42  * AT#ASEND="file.wav",320000
43  ***** Nota:
44  ***** O primeiro parametro é o nome do arquivo a ser amazenado na memoria do modulo.
45  ***** Deve ser escrito junto ao nome a extensão do arquivo "wav".
46  ***** O segundo parametro é o tamanho do arquivo a ser armazenado em bytes. Click com o botão direito no arquivo,
47  ***** va em "Propiedades" na caixa de dialogo do Windows e observe o tamanho exato do arquivo em bytes.
48  *
49  * Após o comando aguarde a resposta do prompt "CONNECT" prompt, depois da resposta, você pode enviar o arquivo de
50  * audio WAV usando a opção "Send File" do ferramente de terminal do Telit AT Controller (use RAW ASCII protocol).
51  * Se a transferência do arquivo for completada com sucesso, o modulo retornara a messagem "OK".
52  * Se o arquivo WAV foi armazenado corretamente, o modulo retorna a messagem "#ASENDEV: 0".
53  *
54  *
55  * * Mais comandos e respostas do modulo que possam ser uteis
56  * AT#ALIST <- Comando: lista os arquivos de audio armazenados no modulo.
57  * #ALIST: <filename>,<filesize>,<crc> <- Parameter: <filename> - file name, string type. <filesize> - file size in bytes
58  * <crc> - CRC16 poly (x^16+x^12+x^5+1) of file in hex format
59  * AT#APLAY=1,1,"file.wav" <- Comando: executa o audio no link do canal de voz.
60  * OK <- Notificação que o comando foi bem suscedido.
61  * #APLAYEV: 0 <- Notificação que o audio chegou ao fim de sua execução.
62  * AT#ADELF="file.wav" <- deleta o arquivo de audio selecionado, armazenado na memoria interna do modulo.
63  * OK
64  *
65  * Note: Arquivos de audio editado podem possivelmente gerar ERRO no momento da execução
66  * ex: (AT#APLAY=1,1,"file.wav") retorna "ERRO" e não executa o audio
67  *
68  * REF:
69  * https://www.telit.com/wp-content/uploads/2017/09/Telit_Audio_Settings_Application_Note__r6.pdf
70  * https://www.telit.com/wp-content/uploads/2017/09/Telit_3G_Modules_AT_Commands_Reference_Guide_r11-1.pdf
71  *
72  */

114 void loop()
115 {
116  char resp[SERIAL_BUFFER_SIZE] = {0};
117  bool call=false;
118  Serial.println("Inicializando...");
119  /*
120  * CONFIG AUTO ANSWER CALL OPTION
121  * //Sets the number of rings required before device automatically answers an incoming call
122  * // n--> number of rings
123  * gsm.RingAutoAnswer(n);
124  */
125  if ( gsm.ConfigureDTMF() ) {
126 
127  Serial.println("Cofiguração DTMF: OK!");
128 
129  delay(10);
130 
131  Serial.println("Esperando ligação!");
132 
133  while (1)
134  {
135  GetBuffer (resp);
136 
137  if (NULL != strstr(resp,"RING")) //A incoming call is indicated returning RING mensage
138  {
139  gsm.ReceiveCall(); //Answer a incoming call
140  call = true;
141  gsm.PlayAudio("menu-dtmf.wav"); //Play audio file stored in the modem, is necessary to write the file name and specify the file extension
142  }
143 
144  while(call==true)
145  {
146  char command = 0;
147  GetBuffer (resp);
148  /*
149  Receiving of a DTMF tone is pointed out with anunsolicited message through
150  AT interface in the following format: #DTMFEV: x with x as the DTMF digit
151  */
152  if (NULL != strstr(resp,"#DTMFEV:"))
153  {
154  for (int i=0; i< sizeof(resp) ; i++)
155  {
156  if (isDigit (resp[i]) )
157  {
158  command = resp[i];
159  break;
160  }
161  }
162  }
163  /*
164  * If there's no active connection it returns NO CARRIER
165  */
166  if (NULL != strstr(resp,"NO CARRIER"))
167  {
168  call = false;
169  Serial.println("Chamada terminada");
170  Serial.println("Esperando ligação!");
171  }
172  switch (command)
173  {
174  case '1':
175  digitalWrite(LED_BUILTIN, HIGH);
176  Serial.println("Led ON!");
177  gsm.PlayAudio("1-dtmf.wav");
178  break;
179  case '2':
180  digitalWrite(LED_BUILTIN, LOW);
181  Serial.println("Led OFF!");
182  gsm.PlayAudio("2-dtmf.wav");
183  break;
184  case '3':
185  gsm.PlayAudio("menu-dtmf.wav");
186  break;
187  default :
188  break;
189  }
190  }
191  }
192  }
